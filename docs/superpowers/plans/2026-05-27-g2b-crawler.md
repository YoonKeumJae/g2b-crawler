# G2B 입찰공고 Crawler Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Node.js CLI that searches 나라장터 입찰공고 for a keyword, paginates all results, visits each detail page, and saves all fields to a CSV file.

**Architecture:** Six focused modules — config, writer, search, paginator, detail, crawler — each with one responsibility. config and writer have Jest unit tests. Playwright modules (search, paginator, detail) each have a headed-mode smoke test step for selector discovery before implementation. The full pipeline is verified by a final end-to-end run.

**Tech Stack:** Node.js 18+ (CommonJS), Playwright (Chromium), Jest

function getDateRange() {
  const to = new Date();
  const from = new Date();
  from.setMonth(from.getMonth() - 1);
  const fmt = (d) => d.toISOString().slice(0, 10).replace(/-/g, '/');
  return { from: fmt(from), to: fmt(to) };
}

};

const fs = require('fs');
const path = require('path');

let headers = null;

function write(outputPath, record) {
  const dir = path.dirname(outputPath);
  if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });

  if (headers === null) {
    headers = Object.keys(record);
    fs.writeFileSync(outputPath, headers.join(',') + '\n');
  }

  const row = headers
    .map((h) => `"${(record[h] || '').toString().replace(/"/g, '""')}"`)
    .join(',');
  fs.appendFileSync(outputPath, row + '\n');
}

function reset() {
  headers = null;
}

module.exports = { write, reset };
```

async function collectUrls(page) {
  const urls = [];

  while (true) {
    await page.waitForLoadState('networkidle');

    // Collect detail links from current page — update selector from Step 1
    const handles = await page.$$('table tbody tr');
    for (const row of handles) {
      const link = await row.$('a');
      if (link) {
        const href = await link.getAttribute('href');
        if (href) urls.push(new URL(href, 'https://www.g2b.go.kr').href);
      }
    }

    // Check for next page — update selector from Step 1
    const nextBtn = await page.$('a.next, button.next');
    if (!nextBtn) break;
    const disabled = await nextBtn.getAttribute('disabled');
    if (disabled !== null) break;

    await nextBtn.click();
  }

  return urls;
}

module.exports = { collectUrls };
```

async function extractDetail(page, url, retries = 3) {
  for (let attempt = 1; attempt <= retries; attempt++) {
    try {
      await page.goto(url, { waitUntil: 'networkidle', timeout: 30000 });
      break;
    } catch (err) {
      if (attempt === retries) {
        console.warn(`[SKIP] Failed after ${retries} attempts: ${url}`);
        return null;
      }
      await new Promise((r) => setTimeout(r, 3000));
    }
  }

  const record = {};

  // Extract all th→td pairs — update selector from Step 1
  const rows = await page.$$('table tr');
  for (const row of rows) {
    const ths = await row.$$('th');
    const tds = await row.$$('td');
    for (let i = 0; i < ths.length; i++) {
      const label = (await ths[i].innerText()).trim();
      const value = tds[i] ? (await tds[i].innerText()).trim() : '';
      if (label) record[label] = value;
    }
  }

  return record;
}

module.exports = { extractDetail };
```

const { chromium } = require('playwright');
const config = require('./config');
const { search } = require('./search');
const { collectUrls } = require('./paginator');
const { extractDetail } = require('./detail');
const { write, reset } = require('./writer');

(async () => {
  const dateRange = config.getDateRange();
  console.log(`Searching: keyword="${config.keyword}", from=${dateRange.from}, to=${dateRange.to}`);

  const browser = await chromium.launch({ headless: config.headless });
  const page = await browser.newPage();

  try {
    await search(page, config.keyword, dateRange);
    console.log('Search submitted. Collecting result URLs...');

    const urls = await collectUrls(page);
    console.log(`Found ${urls.length} announcements. Extracting details...`);

    if (urls.length === 0) {
      console.log('No results found. Exiting.');
      return;
    }

    reset();
    for (let i = 0; i < urls.length; i++) {
      console.log(`[${i + 1}/${urls.length}] ${urls[i]}`);
      const record = await extractDetail(page, urls[i]);
      if (record) write(config.outputPath, record);
    }

    console.log(`Done. Results saved to ${config.outputPath}`);
  } finally {
    await browser.close();
  }
})();
```
